Diretrizes Teóricas Aplicadas:¶

  1. Modelagem de Demandas Voláteis: Avaliação através de Erro Médio Absoluto (MAE), Raiz do Erro Quadrático Médio (RMSE) e Erro Percentual Absoluto Médio (MAPE).

# 6. CAMPEONATO DE MODELOS DE PREVISÃO DE DEMANDA (EIXO I)
print("="*75)
print(" DESEMPENHO DOS MODELOS PREDITIVOS (MÊS DE TESTE: MAIO/2026)")
print("="*75)

melhores_modelos_pcp = {}

for material in df['Material'].unique():
    treino_mat = base_treino[base_treino['Material'] == material].copy()
    teste_mat = base_teste[base_teste['Material'] == material].copy()
    
    y_real = teste_mat['Vol.(TON)'].values
    n_teste = len(teste_mat)
    
    # MODELO A: Média Móvel de 7 dias
    ultimo_valor_treino = treino_mat['Vol.(TON)'].tail(7).mean()
    pred_ma = np.full(n_teste, ultimo_valor_treino)
    
    # MODELO B: Suavização Exponencial Simples (SES)
    model_ses = SimpleExpSmoothing(treino_mat['Vol.(TON)']).fit(smoothing_level=0.3, optimized=False)
    pred_ses = model_ses.forecast(n_teste).values
    
    # MODELO C: Regressão Linear Simples
    X_train = np.array(treino_mat.index).reshape(-1, 1)
    y_train = treino_mat['Vol.(TON)'].values
    X_test = np.array(teste_mat.index).reshape(-1, 1)
    model_lr = LinearRegression().fit(X_train, y_train)
    pred_lr = model_lr.predict(X_test)
    
    modelos = {'Média Móvel': pred_ma, 'Suavização Exp.': pred_ses, 'Regressão Linear': pred_lr}
    erros_material = {}
    
    print(f"\n▶ ANÁLISE DE ACURÁCIA - MATERIAL SKU: {material}")
    
    for nome, pred in modelos.items():
        mae = np.mean(np.abs(y_real - pred))
        rmse = np.sqrt(np.mean((y_real - pred) ** 2))
        mape = np.mean(np.abs((y_real - pred) / np.where(y_real == 0, 1, y_real))) * 100
        
        erros_material[nome] = {'MAE': mae, 'RMSE': rmse, 'MAPE': mape, 'predicoes': pred}
        print(f"  • {nome:16} -> MAE: {mae:6.2f} TON | RMSE: {rmse:6.2f} TON | MAPE: {mape:5.2f}%")
        
    vencedor = min(erros_material, key=lambda k: erros_material[k]['MAPE'])
    
    melhores_modelos_pcp[material] = {
        'modelo_nome': vencedor,
        'pred_valores': erros_material[vencedor]['predicoes'],
        'mae_vencedor': erros_material[vencedor]['MAE'],
        'rmse_vencedor': erros_material[vencedor]['RMSE']
    }
    print(f"  🏆 MODELO SELECIONADO: {vencedor} (Menor margem de erro percentual)")

# 8. ALGORITMO DE SIMULAÇÃO OPERACIONAL E ANÁLISE DE QUEBRAS (BACKTESTING)
print("="*80)
print(" SIMULAÇÃO RETROSPECTIVA DE CENÁRIOS E CONTROLE DE RUPTURAS")
print("="*80)

resumo_simulacao = []

for material in df['Material'].unique():
    teste_mat = base_teste[base_teste['Material'] == material].copy()
    y_real = teste_mat['Vol.(TON)'].values
    dados_prev = melhores_modelos_pcp[material]
    y_pred = dados_prev['pred_valores']
    
    for ns in [0.90, 0.95, 0.99]:
        es_calculado = estoques_seguranca_cenarios[material][ns]
        estoque_atual = es_calculado
        rupturas_detectadas = 0
        volume_nao_atendido = 0
        estoque_medio_acumulado = 0
        
        for t in range(len(teste_mat)):
            demanda_real = y_real[t]
            previsao_modelo = y_pred[t]
            
            # Dinâmica de variação: Estoque absorve o erro real da demanda
            estoque_atual += (previsao_modelo - demanda_real)
            estoque_medio_acumulado += max(0, estoque_atual)
            
            if estoque_atual < 0:
                rupturas_detectadas += 1
                volume_nao_atendido += abs(estoque_atual)
                estoque_atual = 0 # Correção física de contorno fabril
                
        estoque_medio_final = estoque_medio_acumulado / len(teste_mat)
        taxa_atendimento = ((y_real.sum() - volume_nao_atendido) / y_real.sum()) * 100
        
        resumo_simulacao.append({
            'Material': material,
            'Nivel_Servico_Alvo': f"{ns*100:.0f}%",
            'Est_Seguranca': es_calculado,
            'Est_Medio': estoque_medio_final,
            'Rupturas': rupturas_detectadas,
            'Fill_Rate_Real': taxa_atendimento
        })
print("✓ Simulação operacional executada retrospectivamente para Maio/2026.")
